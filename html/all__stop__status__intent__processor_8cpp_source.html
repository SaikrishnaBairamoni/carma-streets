208  void all_stop_status_intent_processor::read_est_path(const rapidjson::GenericArray<true,rapidjson::Value> &est_path, vehicle &vehicle) const {
209 
210 
211  /* the first point in the future path is the current point */
212  std::vector<future_information> future_info;
213  future_information fi;
214  fi.timestamp = vehicle._cur_time;
215  fi.lane_id = vehicle._cur_lane_id;
216  fi.distance = vehicle._cur_distance;
217  future_info.push_back(fi);
218  for (rapidjson::SizeType i = 0; i < est_path.Size(); ++i){
219  rapidjson::GenericObject<true, rapidjson::Value> path_point = est_path[i].GetObject();
220  /* adding checks to make sure the necessary data exist in the future point */
221  // Added member end check for Int64 type
222  if (path_point.FindMember("ts") != path_point.MemberEnd() &&
223  path_point.FindMember("ts")->value.IsUint64() &&
224  path_point.FindMember("id")->value.IsInt() &&
225  path_point.FindMember("ds")->value.IsDouble()){
226 
227  /* adding checks to make sure only valid future points will be saved */
228  if (path_point["id"].GetInt() == 0){
229  SPDLOG_WARN("Ignoring incorrect future point with index {0} for vehicle {1}: \"id\" cannot be 0!", i, vehicle._id );
230  continue;
231  }
232  else if (path_point["ds"].GetDouble() < 0){
233  SPDLOG_WARN("Ignoring incorrect future point with index {0} for vehicle {1}: \"ds\" cannot be negative!", i, vehicle._id );
234  continue;
235  }
236 
237  /* the future path received from CARMA Platform does not consider the stopping requirement at the stop bar.
238  * therefore, CARMA Streets will force the stopping requirement to the vehicle's future path.
239  * basically, if the vehicle is an EV, or and RDV without access, if the vehicle lane id in the future path is not the same as the vehicle entry lane id, the scheduling service will ignore the future points.
240  * */
241  if (future_info[future_info.size() - 1].distance >= path_point["ds"].GetDouble()){
242  fi = future_information();
243  // the unit of timestamp in here is sec with decimal places.
244  fi.timestamp = path_point["ts"].GetInt64();
245  fi.distance = path_point["ds"].GetDouble();
246  fi.lane_id = path_point["id"].GetInt();
247  SPDLOG_DEBUG("future path {0}: {1}, {2}, {3}, {4}", i, fi.lane_id, fi.distance);
248  future_info.push_back(fi);
249  }
250  else{
251  SPDLOG_INFO("Vehicle {0}: the lane id of the vehicle in the est_path has changed!", vehicle._id);
252  break;
253  }
254 
255 
256  }
257  else{
258  throw status_intent_processing_exception("Ignoring incorrect/incomplete future point with index "
259  + std::to_string(i) +" for vehicle " + vehicle._id + "!" );
260  }
261  }
262 
263  vehicle._future_info = future_info;
