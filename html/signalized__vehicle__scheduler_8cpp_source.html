226  uint64_t signalized_vehicle_scheduler::calculate_earliest_entering_time(const streets_vehicles::vehicle &veh) const {
227 
228  if ( veh._cur_state == streets_vehicles::vehicle_state::EV) {
229  // Distance to stop bar
230  double delta_x = veh._cur_distance;
231  // Get Entry Lane
232  OpenAPI::OAILanelet_info entry_lane = get_entry_lanelet_info( veh );
233  // Get Link Lane
234  OpenAPI::OAILanelet_info link_lane = get_link_lanelet_info( veh );
235  // Distance necessary to get to max speed and decelerate with decel_max to departure speed
236  double delta_x_prime = calculate_distance_accel_and_decel( veh, entry_lane.getSpeedLimit(), link_lane.getSpeedLimit() );
237  // Distance necessary to get to the departure speed
238  double delta_x_zegond = calculate_distance_accel_or_decel( veh, link_lane.getSpeedLimit() );
239  SPDLOG_DEBUG("Delta X = {0}, Delta X Prime = {1}, Delta X Zegond = {2}.", delta_x, delta_x_prime, delta_x_zegond);
240 
241  // Calculate v_hat
242  double v_hat = calculate_v_hat(veh, entry_lane.getSpeedLimit(), link_lane.getSpeedLimit(), delta_x, delta_x_prime, delta_x_zegond);
243  SPDLOG_DEBUG("V hat = {0}.", v_hat);
244 
245  // calculate planned acceleration time interval
246  double t_accel = calculate_acceleration_time(veh, v_hat, link_lane.getSpeedLimit(), delta_x, delta_x_zegond);
247  SPDLOG_DEBUG("T accel = {0}.",t_accel);
248 
249  // calculate planned deceleration time interval
250  double t_decel = calculate_deceleration_time(veh, v_hat, link_lane.getSpeedLimit(), delta_x, delta_x_zegond);
251  SPDLOG_DEBUG("T decel = {0}.",t_decel);
252 
253  // Calculate planned cruising time interval
254  double t_cruising = calculate_cruising_time(v_hat, delta_x, delta_x_prime);
255  SPDLOG_DEBUG("T cruising = {0}.",t_cruising);
256 
257  // calculate earliest entering time to stop bar
258  double eet = t_accel + t_cruising + t_decel;
259  if (eet <= 0) {
260  SPDLOG_DEBUG("The estimated eet for vehicle {0} has a negative value = {1}!", veh._id, eet);
261  }
262  return static_cast<uint64_t>(ceil(eet * 1000.0)) + veh._cur_time;
263  }
264  else
265  {
266  SPDLOG_DEBUG("Cannot estimate the earliest entering time for vehicle {0} which is not an EV!", veh._id);
267  throw scheduling_exception("Trying to calculate earliest entering time (EET) for a vehicle that is not an EV!");
268  }
269  return -1;

295  return max_speed;
296  }
297  else if (delta_x_prime > delta_x && delta_x >= delta_x_zegond) {
