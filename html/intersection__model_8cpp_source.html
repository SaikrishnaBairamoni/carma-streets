240  double intersection_model::get_speed_limit_by_lanelet(const lanelet::ConstLanelet &subj_lanelet) const
241  {
242  double speed_limit_result = 0;
243  std::regex r("[0-9]+");
244  lanelet::RegulatoryElementConstPtrs reg_elements = subj_lanelet.regulatoryElements();
245 
246  /***
247  * Identify the speed limit regulatory element by using an attribute with a key value that equals to 'limit'.
248  * The value of this attribute stores the speed limit value in unit of measure MPH
249  **/
250  for (auto itr = reg_elements.begin(); itr != reg_elements.end(); itr++)
251  {
252  lanelet::AttributeMap attr_map = itr->get()->attributes();
253  for (auto inner_itr = attr_map.begin(); inner_itr != attr_map.end(); inner_itr++)
254  {

258  if (std::regex_search(speed_str, r))
259  {
260  speed_limit_result = std::stod(std::regex_replace(speed_str, std::regex("\\s"), std::string("")));

262  };
263  }
264  }

399  if(!map_msg_lane.connection.empty() )

490  void intersection_model::mapping_lane_id_2_lanelet_id(const map_referencepoint& ref_point, const map_lane& lane, const long lane_width, const std::vector<lanelet::ConstLanelet>& subj_lanelets, std::unordered_map<long , lanelet::ConstLanelet>& lane2lanelet_m) const

513  }

521  {

531  //The lane nodes are offsets from the prior node

539  basic_point.y() = cur_y;
